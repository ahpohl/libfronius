57template <typename T> void packInteger(uint16_t *dest, T value) {
58 static_assert(std::is_integral_v<T>, "T must be an integral type");
59
60 constexpr int nBytes = sizeof(T);
61 constexpr int nWords = nBytes / 2;
62
63 unsigned char buf[nBytes];
64 std::memcpy(buf, &value, nBytes);
65
66 for (int i = 0; i < nWords; i++) {
67 unsigned char hi = buf[2 * i + 1];
68 unsigned char lo = buf[2 * i];
69 dest[i] = (static_cast<uint16_t>(hi) << 8) | lo;
70 }
71
72 std::reverse(dest, dest + nWords);
73}

96inline uint32_t modbus_get_uint32(const uint16_t *regs, bool word_swap = false,
97 bool byte_swap = false) {
98 uint16_t hi = regs[0], lo = regs[1];
99 if (byte_swap) {
100 hi = swapBytes16(hi);
101 lo = swapBytes16(lo);
102 }
103 uint32_t val = ((uint32_t)hi << 16) | lo;
104 if (word_swap)
105 val = ((val & 0xFFFF) << 16) | (val >> 16);
106 return val;
107}

128inline uint64_t modbus_get_uint64(const uint16_t *regs, bool word_swap = false,
129 bool byte_swap = false) {
130 uint64_t val = 0;
131 for (int i = 0; i < 4; i++) {
132 uint16_t w = regs[i];
133 if (byte_swap)
134 w = swapBytes16(w);
135 val |= (uint64_t)w << ((3 - i) * 16);
136 }
137 if (word_swap) {
138 val = ((val & 0xFFFFULL) << 48) | ((val & 0xFFFF0000ULL) << 16) |
139 ((val >> 16) & 0xFFFF0000ULL) | ((val >> 48) & 0xFFFFULL);
140 }
141 return val;
142}

186inline FroniusTypes::RemoteEndpoint getSocketInfo(int socket) {
187 struct sockaddr_storage addr;
188 socklen_t addr_len = sizeof(addr);
189
190 if (getpeername(socket, (struct sockaddr *)&addr, &addr_len) != 0) {
191 return {"unknown", 0};
192 }
193
194 char ip_str[INET6_ADDRSTRLEN];
195 int port = 0;
196
197 if (addr.ss_family == AF_INET) {
198 // IPv4
199 struct sockaddr_in *addr_in = (struct sockaddr_in *)&addr;
200 inet_ntop(AF_INET, &addr_in->sin_addr, ip_str, sizeof(ip_str));
201 port = ntohs(addr_in->sin_port);
202 } else if (addr.ss_family == AF_INET6) {
203 // IPv6
204 struct sockaddr_in6 *addr_in6 = (struct sockaddr_in6 *)&addr;
205 inet_ntop(AF_INET6, &addr_in6->sin6_addr, ip_str, sizeof(ip_str));
206 port = ntohs(addr_in6->sin6_port);
207 } else {
208 return {"unknown", 0};
209 }
210
211 return {ip_str, port};
212}

247template <typename T>
248std::expected<void, ModbusError> packToModbus(modbus_mapping_t *dest,
249 Register reg, T value) {
250
251 if (!dest) {
252 return std::unexpected(
253 ModbusError::custom(EINVAL, "Null modbus_mapping_t pointer"));
254 }
255 if (!dest->tab_registers) {
256 return std::unexpected(
257 ModbusError::custom(EINVAL, "modbus_mapping_t has null tab_registers"));
258 }
259
260 switch (reg.TYPE) {
261 case Register::Type::INT16:
262 if constexpr (std::is_integral_v<T>)
263 dest->tab_registers[reg.ADDR] = static_cast<uint16_t>(value);
264 break;
265 case Register::Type::UINT16:
266 if constexpr (std::is_integral_v<T>)
267 dest->tab_registers[reg.ADDR] = value;
268 break;
269 case Register::Type::UINT32:
270 if constexpr (std::is_integral_v<T>)
271 detail::packInteger<uint32_t>(&dest->tab_registers[reg.ADDR], value);
272 break;
273 case Register::Type::UINT64:
274 if constexpr (std::is_integral_v<T>)
275 detail::packInteger<uint64_t>(&dest->tab_registers[reg.ADDR], value);
276 break;
277 case Register::Type::FLOAT:
278 if constexpr (std::is_floating_point_v<T>)
279 modbus_set_float_abcd(static_cast<float>(value),
280 &dest->tab_registers[reg.ADDR]);
281 break;
282 case Register::Type::STRING: {
283 if constexpr (std::is_same_v<T, std::string>) {
284 size_t maxLength = reg.NB * 2;
285 if (value.length() > maxLength) {
286 return std::unexpected(ModbusError::custom(
287 EINVAL,
288 "String length {} exceeds maximum {} characters for register {}",
289 value.length(), maxLength, reg.describe()));
290 }
291
292 // Pack string into modbus mapping
293 for (size_t i = 0; i < value.length() / 2; i++) {
294 unsigned char hi = value[2 * i];
295 unsigned char lo = value[2 * i + 1];
296 dest->tab_registers[reg.ADDR + i] =
297 (static_cast<uint16_t>(hi) << 8) | lo;
298 }
299 if (value.length() % 2) {
300 dest->tab_registers[reg.ADDR + value.length() / 2] =
301 (static_cast<uint16_t>(value[value.length() - 1]) << 8);
302 }
303
304 // Zero out remaining registers
305 for (size_t i = (value.length() + 1) / 2; i < static_cast<size_t>(reg.NB);
306 i++) {
307 dest->tab_registers[reg.ADDR + i] = 0;
308 }
309 }
310 break;
311 }
312 case Register::Type::UNKNOWN:
313 default:
314 return std::unexpected(ModbusError::custom(
315 EINVAL, std::format("Unsupported target register type: {}",
316 Register::typeToString(reg.TYPE))));
317 }
318
319 return {};
320}

354inline std::expected<void, ModbusError> packToModbus(modbus_mapping_t *dest,
355 Register reg, Register sf,
356 double realValue,
357 int decimals) {
358 if (!dest) {
359 return std::unexpected(
360 ModbusError::custom(EINVAL, "Null modbus_mapping_t pointer"));
361 }
362 if (!dest->tab_registers) {
363 return std::unexpected(
364 ModbusError::custom(EINVAL, "modbus_mapping_t has null tab_registers"));
365 }
366
367 // encode float into integer register
368 switch (reg.TYPE) {
369 case Register::Type::INT16: {
370 int16_t value =
371 static_cast<int16_t>(std::round(realValue * std::pow(10, decimals)));
372 dest->tab_registers[reg.ADDR] = static_cast<uint16_t>(value);
373 break;
374 }
375 case Register::Type::UINT16: {
376 uint16_t value =
377 static_cast<uint16_t>(std::round(realValue * std::pow(10, decimals)));
378 dest->tab_registers[reg.ADDR] = value;
379 break;
380 }
381
382 case Register::Type::UINT32: {
383 uint32_t value =
384 static_cast<uint32_t>(std::round(realValue * std::pow(10, decimals)));
385 detail::packInteger<uint32_t>(&dest->tab_registers[reg.ADDR], value);
386 break;
387 }
388 case Register::Type::UINT64: {
389 uint64_t value =
390 static_cast<uint64_t>(std::round(realValue * std::pow(10, decimals)));
391 detail::packInteger<uint64_t>(&dest->tab_registers[reg.ADDR], value);
392 break;
393 }
394
395 default:
396 return std::unexpected(ModbusError::custom(
397 EINVAL,
398 std::format("Unsupported target register type for encoding float: {}",
399 Register::typeToString(reg.TYPE))));
400 }
401
402 // encode scale factor (scale factors are INT16 in SunSpec)
403 int16_t sf_value = -static_cast<int16_t>(decimals);
404 dest->tab_registers[sf.ADDR] = static_cast<uint16_t>(sf_value);
405
406 return {};
407}
