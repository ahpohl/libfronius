43inline uint32_t modbus_get_uint32(const uint16_t *regs, bool word_swap = false,
44 bool byte_swap = false) {
45 uint16_t hi = regs[0], lo = regs[1];
46 if (byte_swap) {
47 hi = swapBytes16(hi);
48 lo = swapBytes16(lo);
49 }
50 uint32_t val = ((uint32_t)hi << 16) | lo;
51 if (word_swap)
52 val = ((val & 0xFFFF) << 16) | (val >> 16);
53 return val;
54}

75inline uint64_t modbus_get_uint64(const uint16_t *regs, bool word_swap = false,
76 bool byte_swap = false) {
77 uint64_t val = 0;
78 for (int i = 0; i < 4; i++) {
79 uint16_t w = regs[i];
80 if (byte_swap)
81 w = swapBytes16(w);
82 val |= (uint64_t)w << ((3 - i) * 16);
83 }
84 if (word_swap) {
85 val = ((val & 0xFFFFULL) << 48) | ((val & 0xFFFF0000ULL) << 16) |
86 ((val >> 16) & 0xFFFF0000ULL) | ((val >> 48) & 0xFFFFULL);
87 }
88 return val;
89}
