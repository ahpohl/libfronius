43template <typename T> void packInteger(uint16_t *dest, T value) {
44 static_assert(std::is_integral_v<T>, "T must be an integral type");
45
46 constexpr int nBytes = sizeof(T);
47 constexpr int nWords = nBytes / 2;
48
49 unsigned char buf[nBytes];
50 std::memcpy(buf, &value, nBytes);
51
52 for (int i = 0; i < nWords; i++) {
53 unsigned char hi = buf[2 * i + 1];
54 unsigned char lo = buf[2 * i];
55 dest[i] = (static_cast<uint16_t>(hi) << 8) | lo;
56 }
57
58 std::reverse(dest, dest + nWords);
59}

82inline uint32_t modbus_get_uint32(const uint16_t *regs, bool word_swap = false,
83 bool byte_swap = false) {
84 uint16_t hi = regs[0], lo = regs[1];
85 if (byte_swap) {
86 hi = swapBytes16(hi);
87 lo = swapBytes16(lo);
88 }
89 uint32_t val = ((uint32_t)hi << 16) | lo;
90 if (word_swap)
91 val = ((val & 0xFFFF) << 16) | (val >> 16);
92 return val;
93}

114inline uint64_t modbus_get_uint64(const uint16_t *regs, bool word_swap = false,
115 bool byte_swap = false) {
116 uint64_t val = 0;
117 for (int i = 0; i < 4; i++) {
118 uint16_t w = regs[i];
119 if (byte_swap)
120 w = swapBytes16(w);
121 val |= (uint64_t)w << ((3 - i) * 16);
122 }
123 if (word_swap) {
124 val = ((val & 0xFFFFULL) << 48) | ((val & 0xFFFF0000ULL) << 16) |
125 ((val >> 16) & 0xFFFF0000ULL) | ((val >> 48) & 0xFFFFULL);
126 }
127 return val;
128}

161inline FroniusTypes::RemoteEndpoint getSocketInfo(int socket) {
162 struct sockaddr_storage addr;
163 socklen_t addr_len = sizeof(addr);
164
165 if (getpeername(socket, (struct sockaddr *)&addr, &addr_len) != 0) {
166 return {"unknown", 0};
167 }
168
169 char ip_str[INET6_ADDRSTRLEN];
170 int port = 0;
171
172 if (addr.ss_family == AF_INET) {
173 // IPv4
174 struct sockaddr_in *addr_in = (struct sockaddr_in *)&addr;
175 inet_ntop(AF_INET, &addr_in->sin_addr, ip_str, sizeof(ip_str));
176 port = ntohs(addr_in->sin_port);
177 } else if (addr.ss_family == AF_INET6) {
178 // IPv6
179 struct sockaddr_in6 *addr_in6 = (struct sockaddr_in6 *)&addr;
180 inet_ntop(AF_INET6, &addr_in6->sin6_addr, ip_str, sizeof(ip_str));
181 port = ntohs(addr_in6->sin6_port);
182 } else {
183 return {"unknown", 0};
184 }
185
186 return {ip_str, port};
187}

217template <typename T>
218std::expected<void, ModbusError> packToModbus(modbus_mapping_t *dest,
219 Register reg, T value) {
220
221 if (!dest) {
222 return std::unexpected(
223 ModbusError::custom(EINVAL, "Null modbus_mapping_t pointer"));
224 }
225 if (!dest->tab_registers) {
226 return std::unexpected(
227 ModbusError::custom(EINVAL, "modbus_mapping_t has null tab_registers"));
228 }
229
230 switch (reg.TYPE) {
231 case Register::Type::INT16:
232 if constexpr (std::is_integral_v<T>)
233 dest->tab_registers[reg.ADDR] = static_cast<uint16_t>(value);
234 break;
235 case Register::Type::UINT16:
236 if constexpr (std::is_integral_v<T>)
237 dest->tab_registers[reg.ADDR] = value;
238 break;
239 case Register::Type::UINT32:
240 if constexpr (std::is_integral_v<T>)
241 detail::packInteger<uint32_t>(&dest->tab_registers[reg.ADDR], value);
242 break;
243 case Register::Type::UINT64:
244 if constexpr (std::is_integral_v<T>)
245 detail::packInteger<uint64_t>(&dest->tab_registers[reg.ADDR], value);
246 break;
247 case Register::Type::FLOAT:
248 if constexpr (std::is_floating_point_v<T>)
249 modbus_set_float_abcd(static_cast<float>(value),
250 &dest->tab_registers[reg.ADDR]);
251 break;
252 case Register::Type::STRING: {
253 if constexpr (std::is_same_v<T, std::string>) {
254 size_t maxLength = reg.NB * 2;
255 if (value.length() > maxLength) {
256 return std::unexpected(ModbusError::custom(
257 EINVAL,
258 "String length {} exceeds maximum {} characters for register {}",
259 value.length(), maxLength, reg.describe()));
260 }
261
262 // Pack string into modbus mapping
263 for (size_t i = 0; i < value.length() / 2; i++) {
264 unsigned char hi = value[2 * i];
265 unsigned char lo = value[2 * i + 1];
266 dest->tab_registers[reg.ADDR + i] =
267 (static_cast<uint16_t>(hi) << 8) | lo;
268 }
269 if (value.length() % 2) {
270 dest->tab_registers[reg.ADDR + value.length() / 2] =
271 (static_cast<uint16_t>(value[value.length() - 1]) << 8);
272 }
273
274 // Zero out remaining registers
275 for (size_t i = (value.length() + 1) / 2; i < static_cast<size_t>(reg.NB);
276 i++) {
277 dest->tab_registers[reg.ADDR + i] = 0;
278 }
279 }
280 break;
281 }
282 case Register::Type::UNKNOWN:
283 default:
284 return std::unexpected(ModbusError::custom(
285 EINVAL, std::format("Unsupported target register type: {}",
286 Register::typeToString(reg.TYPE))));
287 }
288
289 return {};
290}

318inline std::expected<void, ModbusError> packToModbus(modbus_mapping_t *dest,
319 Register reg, Register sf,
320 double realValue,
321 int decimals) {
322 if (!dest) {
323 return std::unexpected(
324 ModbusError::custom(EINVAL, "Null modbus_mapping_t pointer"));
325 }
326 if (!dest->tab_registers) {
327 return std::unexpected(
328 ModbusError::custom(EINVAL, "modbus_mapping_t has null tab_registers"));
329 }
330
331 // encode float into integer register
332 switch (reg.TYPE) {
333 case Register::Type::INT16: {
334 int16_t value =
335 static_cast<int16_t>(std::round(realValue * std::pow(10, decimals)));
336 dest->tab_registers[reg.ADDR] = static_cast<uint16_t>(value);
337 break;
338 }
339 case Register::Type::UINT16: {
340 uint16_t value =
341 static_cast<uint16_t>(std::round(realValue * std::pow(10, decimals)));
342 dest->tab_registers[reg.ADDR] = value;
343 break;
344 }
345
346 case Register::Type::UINT32: {
347 uint32_t value =
348 static_cast<uint32_t>(std::round(realValue * std::pow(10, decimals)));
349 detail::packInteger<uint32_t>(&dest->tab_registers[reg.ADDR], value);
350 break;
351 }
352 case Register::Type::UINT64: {
353 uint64_t value =
354 static_cast<uint64_t>(std::round(realValue * std::pow(10, decimals)));
355 detail::packInteger<uint64_t>(&dest->tab_registers[reg.ADDR], value);
356 break;
357 }
358
359 default:
360 return std::unexpected(ModbusError::custom(
361 EINVAL,
362 std::format("Unsupported target register type for encoding float: {}",
363 Register::typeToString(reg.TYPE))));
364 }
365
366 // encode scale factor (scale factors are INT16 in SunSpec)
367 int16_t sf_value = -static_cast<int16_t>(decimals);
368 dest->tab_registers[sf.ADDR] = static_cast<uint16_t>(sf_value);
369
370 return {};
371}
